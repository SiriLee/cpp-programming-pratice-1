## Trie树

```cpp
#ifndef DSC_TRIE_H
#define DSC_TRIE_H

#include <string>

namespace dsc {

// ============================================================
// Trie — 字典树（前缀树）
// 支持小写字母 a-z，每个节点 26 个子节点 + 结束标记
// ============================================================
class Trie {
private:
    struct Node {
        Node* children_[26];
        bool is_end_;

        Node() : is_end_(false) {
            for (int i = 0; i < 26; ++i) {
                children_[i] = nullptr;
            }
        }
    };

    Node* root_;

    void clear_(Node* node);
    bool remove_(Node* node, const std::string& word, int depth);
    bool is_empty_(Node* node) const;

public:
    Trie() : root_(new Node()) {}
    ~Trie() { clear_(root_); }

    void Insert(const std::string& word);
    bool Search(const std::string& word) const;
    bool StartsWith(const std::string& prefix) const;
    bool Remove(const std::string& word);
};

// ==================== 实现 ====================

inline void Trie::clear_(Node* node) {
    if (!node) return;
    for (int i = 0; i < 26; ++i) {
        clear_(node->children_[i]);
    }
    delete node;
}

inline bool Trie::is_empty_(Node* node) const {
    for (int i = 0; i < 26; ++i) {
        if (node->children_[i]) return false;
    }
    return true;
}

inline void Trie::Insert(const std::string& word) {
    Node* cur = root_;
    for (char ch : word) {
        int idx = ch - 'a';
        if (!cur->children_[idx]) {
            cur->children_[idx] = new Node();
        }
        cur = cur->children_[idx];
    }
    cur->is_end_ = true;
}

inline bool Trie::Search(const std::string& word) const {
    Node* cur = root_;
    for (char ch : word) {
        int idx = ch - 'a';
        if (!cur->children_[idx]) return false;
        cur = cur->children_[idx];
    }
    return cur->is_end_;
}

inline bool Trie::StartsWith(const std::string& prefix) const {
    Node* cur = root_;
    for (char ch : prefix) {
        int idx = ch - 'a';
        if (!cur->children_[idx]) return false;
        cur = cur->children_[idx];
    }
    return true;
}

inline bool Trie::Remove(const std::string& word) {
    return remove_(root_, word, 0);
}

inline bool Trie::remove_(Node* node, const std::string& word, int depth) {
    if (!node) return false;

    if (depth == static_cast<int>(word.size())) {
        if (!node->is_end_) return false;
        node->is_end_ = false;
        return true;
    }

    int idx = word[depth] - 'a';
    Node* child = node->children_[idx];
    if (!child) return false;

    if (!remove_(child, word, depth + 1)) return false;

    if (!child->is_end_ && is_empty_(child)) {
        delete child;
        node->children_[idx] = nullptr;
    }
    return true;
}

} // namespace dsc

#endif // DSC_TRIE_H
```

```cpp
#include "kmp.h"

int* getNext(const std::string& pattern) {
    int len = static_cast<int>(pattern.length());
    int* next = new int[len];
    next[0] = -1; next[1] = 0;
    for (int i = 2; i < len; ++i) {
        int k = next[i - 1];
        while (k >= 0 && pattern[i - 1] != pattern[k]) 
            k = next[k];
        next[i] = k + 1;
    }
    return next;
}

int kmpSearch(const std::string& text, const std::string& pattern) {
    int textLen = static_cast<int>(text.length());
    int patternLen = static_cast<int>(pattern.length());
    if (textLen < patternLen) return -1;
    int* next = getNext(pattern);
    int i = 0, j = 0;
    while (i < textLen && j < patternLen) {
        if (text[i] == pattern[j]) {
            ++i; ++j;
        } 
        else if (j > 0) j = next[j];
        else ++i;
    }
    int p = -1;
    if (j == patternLen) p = i - j;
    delete[] next;
    return p;
}

int* getFixedNext(const std::string& pattern) {
    int* next = getNext(pattern);
    int len = static_cast<int>(pattern.length());
    for (int i = 1; i < len; ++i) {
        if (pattern[i] == pattern[next[i]]) 
            next[i] = next[next[i]];
    }
    return next;
}

int kmpSearchFixed(const std::string& text, const std::string& pattern) {
    int textLen = static_cast<int>(text.length());
    int patternLen = static_cast<int>(pattern.length());
    if (textLen < patternLen) return -1;
    int* next = getFixedNext(pattern);
    int i = 0, j = 0;
    while (i < textLen && j < patternLen) {
        if (text[i] == pattern[j]) {
            ++i; ++j;
        } 
        else if (j > 0) j = next[j];
        else ++i;
    }
    int p = -1;
    if (j == patternLen) p = i - j;
    delete[] next;
    return p;
}
```

## RollingHash

### rolling_hash.h

```cpp
#ifndef ROLLING_HASH_H
#define ROLLING_HASH_H

#include <vector>
#include <string>
#include <utility>

class RollingHash {
public:
    RollingHash(const std::string& s);
    long long getHash(int l, int r) const;
    long long getHash() const;
    std::size_t find(const std::string& pattern) const;
private:
    const int base = 131;
    const int mod = 1e9 + 7;
    std::vector<long long> pre;
    std::vector<long long> powBase;
};


class RollingHashPlus {
public:
    RollingHashPlus(const std::string& s);
    std::pair<long long, long long> getHash(int l, int r) const;
    std::pair<long long, long long> getHash() const;
    std::size_t find(const std::string& pattern) const;
private:
    const int base1 = 131;
    const int mod1 = 1e9 + 7;
    const int base2 = 137;
    const int mod2 = 1e9 + 9;
    std::vector<long long> pre1;
    std::vector<long long> powBase1;
    std::vector<long long> pre2;
    std::vector<long long> powBase2;
};

#endif // ROLLING_HASH_H
```

### rolling_hash.cpp

```cpp
#include "rolling_hash.h"
#include <cmath>

RollingHash::RollingHash(const std::string& s) {
    std::size_t n = s.size();
    pre.resize(n + 1);
    pre[0] = 0;
    powBase.resize(n + 1);
    powBase[0] = 1;
    for (std::size_t i = 0; i < n; ++i) {
        pre[i + 1] = (pre[i] * base + s[i]) % mod;
        powBase[i + 1] = (powBase[i] * base) % mod;
    }
}

long long RollingHash::getHash(int l, int r) const {
    long long hashValue = (pre[r + 1] - pre[l] * powBase[r - l + 1]) % mod;
    if (hashValue < 0) hashValue += mod;
    return hashValue;
}

long long RollingHash::getHash() const {
    return pre.back();
}

std::size_t RollingHash::find(const std::string& pattern) const {
    RollingHash patternHash(pattern);
    long long targetHash = patternHash.getHash();
    std::size_t m = pattern.size();
    for (std::size_t i = 0; i + m <= pre.size() - 1; ++i) {
        if (getHash(i, i + m - 1) == targetHash) return i;
    }
    return std::string::npos;
}

/*---------------------------------------------------------------*/

RollingHashPlus::RollingHashPlus(const std::string& s) {
    std::size_t n = s.size();
    pre1.resize(n + 1); pre2.resize(n + 1);
    pre1[0] = 0; pre2[0] = 0;
    powBase1.resize(n + 1); powBase2.resize(n + 1);
    powBase1[0] = 1; powBase2[0] = 1;
    for (std::size_t i = 0; i < n; ++i) {
        pre1[i + 1] = (pre1[i] * base1 + s[i]) % mod1;
        pre2[i + 1] = (pre2[i] * base2 + s[i]) % mod2;
        powBase1[i + 1] = (powBase1[i] * base1) % mod1;
        powBase2[i + 1] = (powBase2[i] * base2) % mod2;
    }
}

std::pair<long long, long long> RollingHashPlus::getHash(int l, int r) const {
    long long hashValue1 = (pre1[r + 1] - pre1[l] * powBase1[r - l + 1]) % mod1;
    if (hashValue1 < 0) hashValue1 += mod1;
    long long hashValue2 = (pre2[r + 1] - pre2[l] * powBase2[r - l + 1]) % mod2;
    if (hashValue2 < 0) hashValue2 += mod2;
    return {hashValue1, hashValue2};
}

std::pair<long long, long long> RollingHashPlus::getHash() const {
    return {pre1.back(), pre2.back()};
}

std::size_t RollingHashPlus::find(const std::string& pattern) const {
    RollingHashPlus patternHash(pattern);
    auto targetHash = patternHash.getHash();
    std::size_t m = pattern.size();
    for (std::size_t i = 0; i + m <= pre1.size() - 1; ++i) {
        if (getHash(i, i + m - 1) == targetHash) return i;
    }
    return std::string::npos;
}
```
